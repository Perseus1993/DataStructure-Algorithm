这道题可以用数组简单做，这里用的是链表大法，尽量还原了java源码的思想

class MyHashMap {
    MyLinkedList[] set;
    /** Initialize your data structure here. */
    public MyHashMap() {
        set = new MyLinkedList[1000];
        for(int i = 0; i < 1000; i++) {
            set[i] = new MyLinkedList();
        }
    }

    /** value will always be non-negative. */
    public void put(int key, int value) {
        Node node = new Node(key, value);
        int numLinkedList = hashFun(key);
        set[numLinkedList].add(node);
    }

    /** Returns the value to which the specified key is mapped, or -1 if this map contains no mapping for the key */
    public int get(int key) {
        int numLinkedList = hashFun(key);
        return set[numLinkedList].contains(key);
    }

    /** Removes the mapping of the specified value key if this map contains a mapping for the key */
    public void remove(int key) {
        int numLinkedList = hashFun(key);
        set[numLinkedList].remove(key);
    }

    public int hashFun(int i){
        return i % 1000;
    }

}
class MyLinkedList{
    private Node head;
    public void remove(int key){
        if(head == null){
            return;
        }
        Node tmp = head;
        if(head.key == key){
            head = head.next;
        }
        while(tmp.next != null){
            if(tmp.next.key == key){
                tmp.next = tmp.next.next;
            }else{
                tmp = tmp.next;
            }
        }
    }


    public void add(Node node){
        if(head == null){
            head = node;
        }
        Node tmp = head;
        if(this.contains(node.key) == -1){
            while(tmp.next != null){
                tmp = tmp.next;                
            }
            tmp.next = node;
        }else{
           while(tmp.key != node.key){
               tmp = tmp.next;
            }
            tmp.value = node.value;
        }
    }

    public int contains(int key){
        if(head == null){
            return -1;
        }
        Node tmp = head;
        while(tmp != null){
            if(tmp.key == key){
                return tmp.value;
            }else{
                tmp = tmp.next;
            }
        }
        return -1;
    }

}

class Node{
    public int key;
    public int value;
    public Node next;
    public Node(int key, int value){
        super();
        this.key = key;
        this.value = value;
    }
}
